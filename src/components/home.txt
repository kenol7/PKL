export default function Home() {
  // --- State Management ---
  const [semuaRumah, setSemuaRumah] = useState([]);
  const [kotaPengguna, setKotaPengguna] = useState(null);
  const [isLoading, setIsLoading] = useState(true);
  const [slider, setSlider] = useState(null);
  const sliderRef = useRef(null);
  const [slidesPerView, setSlidesPerView] = useState(3);
  

  const endPoint = "https://smataco.my.id/dev/unez/CariRumahAja/api/contribution.php";
  const endpointImage = "https://smataco.my.id/api_digicon/assets/images/";
  const KeyMaps = "AIzaSyDtRAmlhx3Ada5pVl5ilzeHP67TLxO6pyo"; // <-- PENTING: Ganti dengan API Key Anda yang valid

  // --- useEffect Utama untuk Mengambil Semua Data ---
  useEffect(() => {
    const fetchInitialData = async () => {
      setIsLoading(true);
      try {
        const responseRumah = await axios.get(endPoint);
        setSemuaRumah(responseRumah.data);

        if (navigator.geolocation) {
          navigator.geolocation.getCurrentPosition(
            async (position) => {
              const { latitude, longitude } = position.coords;
              
              // Panggil Google Maps Geocoding API menggunakan fetch
              const geoResponse = await fetch(
                `https://maps.googleapis.com/maps/api/geocode/json?latlng=${latitude},${longitude}&key=${KeyMaps}`
              );
              const geoData = await geoResponse.json();

              if (geoData.results && geoData.results.length > 0) {
                // Cari komponen alamat yang merupakan 'locality' (kota) atau 'administrative_area_level_2' (kabupaten/kota)
                const addressComponents = geoData.results[0].address_components;
                const cityComponent = addressComponents.find(
                  (component) => component.types.includes("locality") || component.types.includes("administrative_area_level_2")
                );
                
                if (cityComponent) {
                  const cityName = cityComponent.long_name;
                  console.log("Kota terdeteksi:", cityName);
                  setKotaPengguna(cityName); // Simpan nama kota ke state
                } else {
                  console.log("Nama kota tidak ditemukan dari respons Google Maps.");
                }
              }
            },
            (error) => {
              console.error("Gagal mendapatkan lokasi:", error.message);
              // Jika user menolak izin, tampilkan semua rumah saja
              setKotaPengguna(null);
            }
          );

        console.error("Gagal mengambil data rumah:", error);

    ? semuaRumah.filter((item) => item.city === kotaPengguna)
    : semuaRumah;

  // --- useEffect untuk Slider Responsif ---
  useEffect(() => {
    const updateSlidesPerView = () => {
      const width = window.innerWidth;
      if (width < 768) setSlidesPerView(1);
      else if (width >= 768 && width < 1024) setSlidesPerView(2);
      else setSlidesPerView(3);
    };

    updateSlidesPerView();
    window.addEventListener("resize", updateSlidesPerView);
    return () => window.removeEventListener("resize", updateSlidesPerView);
  }, []);
  
  // --- useEffect untuk Inisialisasi KeenSlider (Slider Atas) ---
  useEffect(() => {
    if (!sliderRef.current) return;
    const newSlider = new KeenSlider(sliderRef.current, {
      loop: true,
      slidesPerView: 1,
      spacing: 10,
      centered: true,
    });
    setSlider(newSlider);
    return () => newSlider.destroy();
  }, []);


  // --- Tampilan Komponen ---
  if (isLoading) {
    return <div className="text-center my-20">Memuat data, mohon tunggu...</div>;
  }

  return (
    <div>
      {/* Bagian Slider Atas */}
      <div className="relative my-10">
        <Search />
        <div className="keen-slider mt-8" ref={sliderRef}>
          {FrameData.map((frame) => (
            <div key={frame.id} className="keen-slider__slide">
              <img src={frame.url} alt={frame.alt} className="w-full h-[500px] object-cover" />
            </div>
          ))}
        </div>
        {slider && (
          <div className="absolute top-1/2 left-0 w-full flex justify-between px-4 -translate-y-1/2">
            <Button onClick={() => slider.prev()} variant="ghost">
              <img src={ArrowLeft} width="30px" height="30px" alt="Previous" />
            </Button>
            <Button onClick={() => slider.next()} variant="ghost">
              <img src={ArrowRight} width="30px" height="30px" alt="Next" />
            </Button>
          </div>
        )}
      </div>

      {/* Bagian Rumah Terdekat */}
      <div className="my-20 mx-10">
        <div className="flex justify-between items-center">
          <div></div>
          <div className="text-center">
            <h3 className="font-semibold text-2xl">Rekomendasi Properti</h3>
            <h3 className="flex justify-center items-center gap-1 mt-1">
              <img src={Location} width="25px" height="25px" alt="Location" />
              <span>{kotaPengguna || "Seluruh Wilayah"}</span>
            </h3>
          </div>
          <div>
            <a href="#" className="bg-yellow-200 px-5 py-1.5 rounded-lg shadow-md">
              Lihat Semua
            </a>
          </div>
        </div>
        
        <div className="flex justify-center mt-12">
          {filteredRumahTerdekat.length === 0 ? (
            <p className="text-gray-600 text-lg text-center">
              Tidak ada rumah ditemukan di wilayah Anda saat ini.
            </p>
          ) : (
            <Swiper
              effect="coverflow"
              grabCursor={true}
              centeredSlides={true}
              loop={true}
              slidesPerView={slidesPerView}
              coverflowEffect={{
                rotate: 0,
                stretch: 0,
                depth: 100,
                modifier: 2.5,
                slideShadows: false,
              }}
              modules={[EffectCoverflow, Navigation]}
              className="w-full"
            >
              {filteredRumahTerdekat.map((item) => (
                <SwiperSlide key={item.id}>
                  <div className="flex justify-center">
                    <HouseCard data={item} imageBaseUrl={endpointImage} />
                  </div>
                </SwiperSlide>
              ))}
            </Swiper>
          )}
        </div>
      </div>

      {/* Bagian lain dari halaman Anda bisa ditambahkan di sini */}
    </div>
  );
}
